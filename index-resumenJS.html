¿Qué es una variable?

// Una variable es un contenedor para almacenar datos.
let nombre = "Juan";

¿Cómo se declara una variable?

// Se declara una variable con let, var, o const.
let edad = 25;

var, let, const

// var: Alcance global o de función, puede ser redeclarada.
var ciudad = "Madrid";
// let: Alcance de bloque, no puede ser redeclarada en el mismo bloque.
let pais = "España";
// const: Alcance de bloque, valor constante.
const fecha = "2024";

Variable numérica y cadena de texto

// Variable numérica
let edad = 30;
// Variable cadena de texto
let nombre = "Ana";

Operadores

Operadores aritméticos

// Sumar, restar, multiplicar, dividir
let suma = 5 + 3;
let resta = 5 - 3;
let multiplicacion = 5 * 3;
let division = 5 / 3;

Operadores de comparación

// Comparar valores
let esIgual = (5 === 5);
let esMayor = (5 > 3);

Operadores lógicos

// AND, OR, NOT
let y = (true && false);
let o = (true || false);
let not = !true;

Estructuras de control

Bucles

// Bucle for
for (let i = 0; i < 5; i++) {
    console.log(i);
}
// Bucle while
let i = 0;
while (i < 5) {
    console.log(i);
    i++;
}
// Bucle do while
let j = 0;
do {
    console.log(j);
    j++;
} while (j < 5);

Condicionales

if

// Condición simple
if (edad > 18) {
    console.log("Adulto");
}

switch

// Evaluar múltiples casos
let dia = 2;
switch (dia) {
    case 1:
        console.log("Lunes");
        break;
    case 2:
        console.log("Martes");
        break;
    default:
        console.log("Otro día");
}

¿Qué es un array?

// Un array es una lista de elementos
let frutas = ["manzana", "naranja", "plátano"];

¿Qué es un booleano?

// Un booleano representa verdadero o falso
let esVerano = true;

Funciones

¿Qué es una función?

// Una función es un bloque de código reutilizable
function saludar() {
    console.log("Hola");
}

¿Cómo se declara una función?

// Declaración de una función con parámetros
function sumar(a, b) {
    return a + b;
}

Parámetros y argumentos

// Llamada a la función
let resultado = sumar(5, 3); // resultado es 8

Objetos

¿Qué es un objeto?

// Un objeto es una colección de propiedades
let persona = {
    nombre: "Ana",
    edad: 30,
    saludar: function() {
        console.log("Hola, soy " + this.nombre);
    }
};

Arrays de objetos

// Un array de objetos es una lista de objetos
let personas = [
    { nombre: "Ana", edad: 30 },
    { nombre: "Luis", edad: 25 }
];
